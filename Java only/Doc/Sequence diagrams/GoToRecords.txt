title Event: GoToRecords

participant Request
participant GoToRecords
participant Session
participant ExamRecordDAO
participant ctx
participant records.html
participant Response

Request->GoToRecords: doGet()

GoToRecords->Session: getAttribute("user")
GoToRecords->ExamRecordDAO: findExamRecordsByProfessorUUID(userUUID)
ExamRecordDAO->GoToRecords: erList

alt erList == null or empty
    GoToRecords->Response: sendError(400, "no records yet")
else
    GoToRecords->GoToRecords: erList.stream().sorted(by course name, exam date).toList()
    GoToRecords->ctx: setVariable("records", sorted)
    GoToRecords->records.html: templateEngine.process(path, ctx, response.getWriter())
end
