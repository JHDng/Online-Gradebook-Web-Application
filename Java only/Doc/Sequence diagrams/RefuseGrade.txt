title Event: RefuseGrade

participant Request
participant RefuseGrade
participant ExamDateDAO
participant GoToExamResult
participant Response

Request->RefuseGrade: doPost(examDateID)

alt !isStringSafe(examDateID_str)
    RefuseGrade->Response: sendError("Invalid exam date id")
end

alt examDateID_str is not a number
    RefuseGrade->Response: sendError("Invalid exam date id format")
end

RefuseGrade->ExamDateDAO: refuseGrade(studentID, examDateID)


    RefuseGrade->GoToExamResult: response.sendRedirect
